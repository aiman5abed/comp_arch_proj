; opcode=0, rd=2, rs=0, rt=1, imm=0x00A

; (0<<24)|(4<<20)|(2<<16)|(3<<12)|0 = 0x00423000

; (1<<24)|(5<<20)|(2<<16)|(3<<12)|0 = 0x01523000

; (0<<24)|(7<<20)|(0<<16)|(1<<12)|0x00F = 0x0070100F

; PC=5: R6 = R2 AND R7 -> R6 = 10 AND 15 = 10
; (2<<24)|(6<<20)|(2<<16)|(7<<12)|0 = 0x02627000

; (5<<24)|(8<<20)|(2<<16)|(3<<12)|0 = 0x05823000
